7. Stats auto-wiring — update map, homepage syncs

The new map file broadcasts its aggregated stats (active signals, tracked value, municipalities, last sync) via postMessage every load. The homepage listens and updates the stat cards. You only edit the map signals array — homepage totals update automatically.

/* on mobile stack vertically */
@media (max-width: 640px){ .stat-row{ grid-template-columns:1fr; } }
</style>

JS — paste once in your homepage (near end of body)

<script>
// Listen for stats broadcast from the map iframe
window.addEventListener('message', (e) => {
  // Only trust messages from the map origin
  if (e.origin !== 'https://ftl-signal.github.io') return;
  if (!e.data || e.data.type !== 'signalStats') return;
  const s = e.data;

  const set = (key, val) => {
    const el = document.querySelector(`[data-stat="${key}"]`);
    if (el && val != null) el.textContent = val;
  };
  set('active', s.activeSignals);
  set('value',  s.trackedValueLabel);
});

// Handshake — ask the map for stats once the iframe is loaded
window.addEventListener('load', () => {
  const frame = document.querySelector('.map-iframe');
  if (frame && frame.contentWindow) {
    try { frame.contentWindow.postMessage({type:'requestSignalStats'}, '*'); }
    catch(err){}
  }
});
</script>

Flow Edit signals in the map repo → push → iframe reloads with new data → map posts new totals → homepage cards update. You literally only maintain the map file.

JS — sunrise/sunset by local FL month (accurate to ~15 min)

<script>
(function(){
  // Pompano Beach FL — hour of day (0-24, local time)
  // DST-aware because getHours() returns local wall clock
  const MONTHLY_SUN = [
    {rise:7.12, set:17.77, m:'Jan'},
    {rise:6.87, set:18.10, m:'Feb'},
    {rise:7.48, set:19.35, m:'Mar'}, // DST begins mid-month
    {rise:6.92, set:19.60, m:'Apr'},
    {rise:6.50, set:19.87, m:'May'},
    {rise:6.40, set:20.17, m:'Jun'},
    {rise:6.57, set:20.20, m:'Jul'},
    {rise:6.82, set:19.92, m:'Aug'},
    {rise:7.07, set:19.37, m:'Sep'},
    {rise:7.30, set:18.78, m:'Oct'},
    {rise:6.62, set:17.43, m:'Nov'}, // DST ends early-month
    {rise:6.95, set:17.47, m:'Dec'}
  ];
  function check(){
    const now = new Date();
    const hour = now.getHours() + now.getMinutes()/60;
    const {rise, set} = MONTHLY_SUN[now.getMonth()];
    const isDay = hour >= rise && hour < set;
    const frames = document.querySelectorAll('[data-cam]');
    frames.forEach(f => {
      if (isDay) f.removeAttribute('data-cam-night');
      else      f.setAttribute('data-cam-night', '');
    });
    // Show estimated sunrise under the moon
    const riseDate = new Date(now);
    if (hour >= set) riseDate.setDate(riseDate.getDate()+1);
    const rh = Math.floor(rise), rm = Math.round((rise-rh)*60);
    const riseTxt = `Est. sunrise ${rh}:${String(rm).padStart(2,'0')} AM`;
    document.querySelectorAll('[data-sunrise]').forEach(el => el.textContent = riseTxt);
  }
  check();
  setInterval(check, 60*1000); // re-check every minute
})();
</script>

Bonus If you ever move cameras or want minute-accurate times, swap the table for SunCalc.js (3 KB, from npm). The rest of the logic is unchanged.
