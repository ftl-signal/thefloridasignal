function makeCircleIcon(color, size = 16) {

    return L.divIcon({

        className: '',

        html: `<div style="

            width:${size}px;height:${size}px;

            background:${color};border-radius:50%;

            border:2px solid rgba(255,255,255,0.18);

            box-shadow:0 0 10px ${color}99,0 0 22px ${color}44;

            cursor:pointer;transition:transform 0.15s;"

            onmouseover="this.style.transform='scale(1.6)'"

            onmouseout="this.style.transform='scale(1)'">

        </div>`,

        iconSize: [size, size],

        iconAnchor: [size / 2, size / 2]

    });

}


function makeDiamondIcon(color, size = 16) {

signals.forEach(sig => {

    if (sig.shape === 'diamond') {

        // Dashed border for area/approx pins, solid for exact

        icon = (sig.parcelNote)

            ? makeAreaDiamondIcon(sig.color)

            : makeDiamondIcon(sig.color);

    } else {

        icon = makeCircleIcon(sig.color);

    }


    const marker = L.marker([sig.lat, sig.lng], { icon })

        .addTo(layerGroups[sig.layer])

        .on('click', () => openSidebar(sig));


    // Tooltip: add ⊙ flag for approximate coords

    const approxFlag = sig.coordApprox

        ? `<br><span style="color:#c9a84c;font-size:11px;">⊙ coordinates approximate</span>`

        : '';

    const parcelFlag = sig.parcelNote

        ? `<br><span style="color:#55cc99;font-size:11px;">◎ no parcel named yet — area pin</span>`

        : '';


    marker.bindTooltip(

        `<strong>${sig.name}</strong><br>

         <span style="color:#888;font-size:12px;">${sig.address}</span>

         ${approxFlag}${parcelFlag}`,

        { direction: 'top', offset: [0, -12], permanent: false }

    );

});

function openSidebar(sig) {

    document.getElementById('s-tier').textContent    = sig.tierLabel;

    document.getElementById('s-tier').className      = `s-tier ${sig.tierClass}`;

    document.getElementById('s-name').textContent    = sig.name;

    document.getElementById('s-address').textContent = sig.address;


    // Build flags under the address

    let flagHTML = '';

    if (sig.coordApprox && !sig.parcelNote) {

        flagHTML += `<div class="coord-flag">⊙ Coordinates approximate — street-level placement, not parcel-corner-exact</div>`;

    }

    if (sig.parcelNote) {

        flagHTML += `<div class="coord-flag">⊙ Area pin — no specific parcel named in source document</div>`;

    }

    document.getElementById('s-flags').innerHTML = flagHTML;


    // Permit data table

    let rowsHTML = sig.rows.map(r =>

        `<tr><td>${r[0]}</td><td>${r[1]}</td></tr>`

    ).join('');

    let html = `<table class="permit-table">${rowsHTML}</table>`;


    // Parcel precision note (FATVillage etc.)

    if (sig.parcelNote) {

        html += `

            <div class="section-label">Parcel Precision Note</div>

            <div class="inset-box inset-teal">${sig.parcelNote}</div>`;

    }


    // Extra detail

    if (sig.extra) {

        html += `

            <div class="section-label">Additional Record Detail</div>

            <div class="inset-box">${sig.extra}</div>`;

    }


    // What to Watch

    html += `

        <div class="section-label">What to Watch</div>

        <div class="inset-box inset-blue">${sig.watch}</div>

        <span class="signal-pill ${sig.signalClass}">${sig.signal}</span>`;


    // Source links

    html += `<div class="source-links">`;

    html += `<a href="${sig.bcpaUrl}" target="_blank" class="record-link">

        🔗 &nbsp;BCPA Property Search →

    </a>`;

    html += `<a href="${sig.portalUrl}" target="_blank" class="record-link">

        🔗 &nbsp;${sig.portalLabel}

    </a>`;

    html += `</div>`;


    document.getElementById('sidebar-body').innerHTML = html;

    document.getElementById('sidebar').classList.add('open');

    map.panTo([sig.lat, sig.lng], { animate: true, duration: 0.4 });

}


function closeSidebar() {

    document.getElementById('sidebar').classList.remove('open');

}

function toggleLayer(name) {

    layerState[name] = !layerState[name];

    const toggle = document.getElementById(`toggle-${name}`);

    if (layerState[name]) {

        map.addLayer(layerGroups[name]);

        toggle.classList.add('on');

    } else {

        map.removeLayer(layerGroups[name]);

        toggle.classList.remove('on');

    }

}
